% -----------------------------------------------------------------
% Slide 4: System Architecture
% -----------------------------------------------------------------
\begin{frame}{System Architecture \& Data Flow}
  \begin{block}{Data Processing Pipeline}
    \begin{enumerate}
      \item \textbf{Event Ingestion}: Python Event Producer publishes transactional events to a local Apache Kafka broker topic.
      \item \textbf{Streaming Ingestion}: Spark Structured Streaming consumes events incrementally, parses payloads, and checks quality.
      \item \textbf{Isolation Storage}: Valid events are saved to \texttt{data/raw/}; rejected events go to \texttt{data/rejects/}.
      \item \textbf{Workflow Orchestration}: Apache Airflow coordinates batch jobs, executing daily summaries and validating downstream files.
    \end{enumerate}
  \end{block}
\end{frame}

% -----------------------------------------------------------------
% Slide 5: Event Data Contract \& Domain
% -----------------------------------------------------------------
\begin{frame}{Event Data Contract \& Domain}
  \begin{block}{Retail Inventory Management Domain}
    Using transactional inventory events (\texttt{inventory.stock\_change}) generated by the streamflow producer.
  \end{block}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Envelope Schema}
        \begin{itemize}
          \item \texttt{event\_id}: String (UUID)
          \item \texttt{event\_type}: String
          \item \texttt{event\_ts}: ISO 8601 Timestamp
          \item \texttt{source}: Identifier
          \item \texttt{payload}: Nested struct
        \end{itemize}
      \end{block}
    \end{column}
    
    \hfill
    
    \begin{column}{0.48\textwidth}
      \begin{block}{Payload Details}
        \begin{itemize}
          \item \texttt{sku\_id}: Product ID
          \item \texttt{quantity\_delta}: Int
          \item \texttt{current\_stock\_level}: Int
          \item \texttt{checkout\_lane}: Nullable String
          \item \texttt{price}: Double
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

% -----------------------------------------------------------------
% Slide 6: Stream Ingestion & Checkpointing
% -----------------------------------------------------------------
\begin{frame}{Stream Ingestion \& Checkpointing}
  \begin{block}{Spark Structured Streaming (\texttt{streaming\_ingest.py})}
    \begin{itemize}
      \item Consumes micro-batches incrementally from the Apache Kafka broker.
      \item Integrates externalized configs (bootstrap servers, topic filters) loaded from \texttt{pipeline.yml} or environment variables.
      \item Outputs processed records directly to high-performance Parquet format.
    \end{itemize}
  \end{block}

  \begin{exampleblock}{Fault Tolerance \& Durability}
    \begin{itemize}
      \item Utilizes Spark DFS checkpointing (\texttt{data/checkpoints/raw}) to track read progress.
      \item Provides state recovery to ensure exactly-once ingestion semantics upon container restart.
    \end{itemize}
  \end{exampleblock}
\end{frame}

% -----------------------------------------------------------------
% Slide 7: Data Quality & Rules Engine
% -----------------------------------------------------------------
\begin{frame}{Data Quality \& Rules Engine}
  \begin{block}{Validation Framework (\texttt{quality.py})}
    Performs checks at the processing edge before storage:
    \begin{itemize}
      \item \textbf{Structure}: Non-empty fields, valid JSON parse.
      \item \textbf{Constraints}: Match allowed sources/event types, valid ISO timestamps.
      \item \textbf{Deduplication}: Identifies duplicates in active micro-batches and joins against historical raw logs to prevent double-processing.
    \end{itemize}
  \end{block}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Valid Stream (Parquet)}
        Clean, schema-compliant events written directly to \texttt{data/raw/}.
      \end{block}
    \end{column}
    
    \hfill
    
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{Rejected Stream (Audit)}
        Failing records written to \texttt{data/rejects/} with appended failure reasons.
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

% -----------------------------------------------------------------
% Slide 8: Workflow Orchestration (Apache Airflow)
% -----------------------------------------------------------------
\begin{frame}{Workflow Orchestration}
  \begin{block}{DAG: \texttt{streamflow\_daily\_summary}}
    Airflow coordinates bounded batch aggregation cycles (no infinite streaming loops inside orchestrator).
  \end{block}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Step 1: Aggregation Job}
        \textbf{SparkSubmitOperator}:
        \begin{itemize}
          \item Triggers batch job \texttt{daily\_summary.py}.
          \item Summarizes metrics from raw Parquet data.
          \item Overwrites curated analytical summaries.
        \end{itemize}
      \end{block}
    \end{column}
    
    \hfill
    
    \begin{column}{0.48\textwidth}
      \begin{block}{Step 2: Output Validation}
        \textbf{PythonOperator}:
        \begin{itemize}
          \item Validates curated directories exist.
          \item Checks output files are non-empty.
          \item Emits job status and metadata.
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

% Slide 10: Engineering Challenges
% -----------------------------------------------------------------
\begin{frame}{Engineering Challenges}
  \begin{block}{Key Architecture Bottlenecks Resolved}
    \begin{itemize}
      \item \textbf{Docker Networking}: Configured dual-listener setup inside Compose for Kafka (\texttt{localhost:9092} for host vs. internal container routing via \texttt{kafka:29092}).
      \item \textbf{DataFrame Schema Parsing}: Safely flattened multi-nested JSON structures directly inside PySpark streaming using explicit spark schema structures.
      \item \textbf{DFS Checkpoint Permissions}: Resolved local folder mount read/write access and umask restrictions under Dockerized Spark executions.
    \end{itemize}
  \end{block}
\end{frame}
